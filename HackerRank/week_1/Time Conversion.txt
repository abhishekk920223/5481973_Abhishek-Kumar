string timeConversion(string s) {
    string period = s.substr(8, 2);
    int hour = stoi(s.substr(0, 2));
    if (period == "AM" && hour == 12)
    {
        hour = 0;
    } 
    if (period == "PM" && hour != 12)
    {
        hour += 12;
    } 

    // use stringstream
    stringstream ss;
    ss << setfill('0') << setw(2) << hour << s.substr(2, 6);
    return ss.str();
}
