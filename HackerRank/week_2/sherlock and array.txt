string balancedSums(vector<int> arr) {
    long long total = 0;
    for (int x : arr) total += x;

    long long leftSum = 0;
    for (int i = 0; i < arr.size(); i++) {
        long long rightSum = total - leftSum - arr[i];
        if (leftSum == rightSum) {
            return "YES";
        }
        leftSum += arr[i];
    }
    return "NO";
}
