int cookies(int k, vector<int> A) {
    priority_queue<int, vector<int>, greater<int>> minHeap(A.begin(), A.end());
    int operations = 0;

    while (!minHeap.empty() && minHeap.top() < k) {
        if (minHeap.size() < 2) return -1;  // not enough cookies to combine

        int least = minHeap.top(); minHeap.pop();
        int second = minHeap.top(); minHeap.pop();

        int newCookie = least + 2 * second;
        minHeap.push(newCookie);
        operations++;
    }
    return operations;
}
