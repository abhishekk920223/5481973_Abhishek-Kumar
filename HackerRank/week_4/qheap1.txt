#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int Q;
    cin >> Q;

    priority_queue<int, vector<int>, greater<int>> minHeap;
    unordered_map<int, int> deleted;  // keeps track of elements to delete

    while (Q--) {
        int type;
        cin >> type;

        if (type == 1) {
            int x;
            cin >> x;
            minHeap.push(x);
        } 
        else if (type == 2) {
            int x;
            cin >> x;
            deleted[x]++;  // mark as deleted
        } 
        else if (type == 3) {
            // remove all invalid (deleted) tops
            while (!minHeap.empty() && deleted[minHeap.top()] > 0) {
                deleted[minHeap.top()]--;
                minHeap.pop();
            }
            cout << minHeap.top() << "\n";
        }
    }
    return 0;
}
