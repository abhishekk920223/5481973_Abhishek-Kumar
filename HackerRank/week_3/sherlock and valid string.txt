string isValid(string s) {
    unordered_map<char, int> freq;
    for (char c : s) {
        freq[c]++;
    }

    unordered_map<int, int> countFreq; // frequency -> how many chars have it
    for (auto &p : freq) {
        countFreq[p.second]++;
    }

    if (countFreq.size() == 1) {
        return "YES"; // all frequencies equal
    }
    if (countFreq.size() == 2) {
        auto it = countFreq.begin();
        int f1 = it->first, c1 = it->second;
        it++;
        int f2 = it->first, c2 = it->second;

        // Case 1: one frequency is 1 and it appears once
        if ((f1 == 1 && c1 == 1) || (f2 == 1 && c2 == 1)) {
            return "YES";
        }
        // Case 2: difference between freqs is 1 and higher one appears once
        if ((abs(f1 - f2) == 1) && ((f1 > f2 && c1 == 1) || (f2 > f1 && c2 == 1))) {
            return "YES";
        }
    }
    return "NO";
}
